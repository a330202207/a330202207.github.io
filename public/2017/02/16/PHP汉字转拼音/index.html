    /**
     * 将中文编码成拼音
     * @param string $str 要转换为拼音的字符串
     * @param string $RetFormat 返回格式 [first:每个字的首字母|all:全拼音|one:字符串字母]
     * @return string
     */
    public function get_pinyin($str, $RetFormat='first')
    {
        $GBK = iconv('UTF-8', 'GBK', $str);
        $UTF8 = iconv('GBK', 'UTF-8', $GBK);
        if($UTF8 != $str) $GBK = $str;
        $Buf = array();
        for ($i=0, $iLoop=strlen($GBK); $i<$iLoop; $i++) {
            $Chr = ord($GBK{$i});
            if ($Chr>160) {
                $Chr = ($Chr<<8) + ord($GBK{++$i}) - 65536;
            }
            if ('first' == $RetFormat || 'one' == $RetFormat) {
                $Buf[] = substr($this->zh2py($Chr),0,1);
            }else{
                $Buf[] = $this->zh2py($Chr);
            }

        }
        if ('first' === $RetFormat) {
            return implode('', $Buf);
        }elseif('one' == $RetFormat) {
            return $Buf[0];
        }else{
            return implode(' ', $Buf);
        }
    }

    /**
     * 中文转换到拼音(每次处理一个字符)
     * @param number $iWORD 待处理字符双字节
     * @return string 拼音
     */
    private function zh2py($iWORD) 
    {
        if($iWORD>0 && $iWORD<160 ) {
            return chr($iWORD);
        } elseif ($iWORD<-20319||$iWORD>-10247) {
            return '';
        } else {
            foreach ($this->data as $py => $code) {
                if($code > $iWORD) break;
                $result = $py;
            }
            return $result;
        }
    }
